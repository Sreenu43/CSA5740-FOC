#include <stdio.h>

int main()
{
    int array[100], n, sum = 0;
    printf("Enter the number of elements: ");
    scanf("%d", &n);
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &array[i]);
    }
    for (int i = 0; i < n; i++) {
        sum += array[i];
    }
    printf("Sum of elements: %d\n", sum);
    return 0;
}
